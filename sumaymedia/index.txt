    <script>
        function conseguirSuma(suma = 0, contador = 0){
            nmr = Number (prompt("Introduce un número... (Negativo para terminar)"));
            if (nmr < 0){
                alert(`La suma es: ${suma}`);
                operar(suma, contador);
            } else{
                suma = suma + nmr;
                contador += 1;
                conseguirSuma(suma, contador);
            }
        }
        function operar(suma, contador){
            alert("La media de la operación es: " + suma/contador); 
        }
        conseguirSuma();
    </script>
